self.apellido = apellido

return self.nombre

import Persona

self.titulo = titulo

return self.cantHijos
def set_cantHijos(self, cantHijos):
self.cantHijos = cantHijos
def get_titulo(self):
return self.titulo
def set_titulo(self, titulo):

Alumno.py:
